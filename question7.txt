#include <ctype.h>
#include <stack.h>
#include <queue.h>

char* saida_polonesa(char* expressao){

  stack<char> operadores
  queue<char> letras;
  char* saida = "";

  int i = 0;
  while( expressao[i] != '\0' ){

    if( isalpha(expressao[i]) ) 
      letras.push(expressao[i]);
    else if( expressao[i] != '(' && expressao[i] != ')' )    
      operadores.push(expressao[i]);

    if(expressao[i] == ')'){
      while(!letras.empty()){
        saida += letras.front();
        letras.pop();
      }
      while(!operadores.empty()){
        saida += operadores.top();
        operadores.pop();
      }
    } 

    i++;
  }

  return saida;
}
